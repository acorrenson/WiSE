Definition bijection (A B : Type) `{Eq A} `{Eq B} (f : A -> B) (g : B -> A) :=
  (forall a, g (f a) ≡ a) /\ (forall b, f (g b) ≡ b).

Definition isomorph (A B : Type) `{Eq A} `{Eq B} :=
  exists f g, bijection A B f g.
